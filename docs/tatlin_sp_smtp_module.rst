Examples
--------

.. code-block:: yaml+jinja

    
    ---
    - name: Set SMTP config
      yadro.tatlin_uni.tatlin_sp_smtp:
        connection: "{{ connection }}"
        address: 127.0.0.1
        port: 25
        login: user
        password: userpass
        sender: smtp@example.com
        state: present
        recipients:
          - first@recipient.com
          - second@recipient.com

    - name: Add recipients
      yadro.tatlin_uni.tatlin_sp_smtp:
        connection: "{{ connection }}"
        recipients:
          - first@recipient.com
          - second@recipient.com
          - third@recipient.com

    - name: Remove recipient
      yadro.tatlin_uni.tatlin_sp_smtp:
        connection: "{{ connection }}"
        recipients:
          - second@recipient.com
        state: absent

    - name: Clear config
      yadro.tatlin_uni.tatlin_sp_smtp:
        connection: "{{ connection }}"
        state: absent
