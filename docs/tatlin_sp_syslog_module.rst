.. Collection note

.. note::
    This module is part of the `yadro.tatlin_uni collection <https://galaxy.ansible.com/yadro/tatlin_uni>`_ (version 1.0.0).

Examples
--------

.. code-block:: yaml+jinja

    
    ---
    - name: Set Syslog recipients
      yadro.tatlin_uni.tatlin_sp_syslog:
        connection: "{{ connection }}"
        recipients:
          - address: 127.0.0.1
            port: 514
            protocol: udp
            facility: 10
            severity: critical
            audit: false
          - address: example.com
            port: 601
            protocol: tls
            facility: 15
            severity: warning
            audit: true
        state: present

    - name: Remove recipient by address
      yadro.tatlin_uni.tatlin_sp_syslog:
        connection: "{{ connection }}"
        recipients:
          - address: 127.0.0.1
        state: absent

    - name: Remove recipient by address and port
      yadro.tatlin_uni.tatlin_sp_syslog:
        connection: "{{ connection }}"
        recipients:
          - address: 127.0.0.1
            port: 514
        state: absent

    - name: Reset config
      yadro.tatlin_uni.tatlin_sp_syslog:
        connection: "{{ connection }}"
        state: absent
